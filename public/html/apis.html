Send Raw Transaction

POST
Request 
         YOUR_DOMAIN/api/v1/sendrawtx
         // sendrawtx takes two parameters, rawTx, and checkInputs (optional)
         // use x-www-form-urlencoded & surround rawTx with quotes
         rawTx: string and checkInputs: integer
Response 
// Pass 
{
  "success": true,
  "status": "sent"
}

// Fail 
{
  "success": false,
  "error": "Transaction hex string decode failed"
}

Websocket Channel

Connection URL 
         wss://bcblockexplorer.com/
Response 
        // Correctly connecting to the websocket channel
        {"message":"B&C Explorer Websocket connection established"}
        // Once your connected, the websocket server can broadcast the following

        // messages are sent in JSON format, so make sure to JSONparse([object]) 
        //sent to clients when new block found
        {"type": "status", "info": [object]}
        //also sent to clients when new block found 
        {"type": "block", "info": [object]}
        //sent to clients when unconfirmed tx. found
        {"type": "tx", "info": [object]}
        //sent to clients when motion is found or updated
        {"type": "motion", "op":motion_data.type,"info": [object]} 
        //sent to clients when custodian is found or updated
        {"type": "custo", "op":custo_data.type, "info": [object]}
